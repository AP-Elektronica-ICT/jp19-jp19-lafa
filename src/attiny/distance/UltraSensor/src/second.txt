#include <Arduino.h>
#include <NewPing.h>

int maxDIstance = 100, distance;
NewPing sonar (PB1,PB2,maxDIstance);

void setup(){
Serial.begin(9600);
//D9 as output
//Trigger pin   
DDRB |= ( 1 << PB1);
//D10 as input
//Echo pin
DDRB |= ~( 1 << PB2);
}

void loop(){
distance = sonar.ping_cm();
delay(50);
Serial.println(distance);
}
